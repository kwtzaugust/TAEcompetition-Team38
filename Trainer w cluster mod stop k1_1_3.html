        n = dim(eval(parse(text = dataname)))[1]

        prediction = rep(guess, n)
        names = rownames(eval(parse(text = dataname)))
        names(prediction) = names
        assign(predname, prediction)

total_correct = 0
total_predicted = 0

for (k in 1:centers) {
    num_correct = 0
    
    prediction = eval(parse(text = paste0("predict",k)))
    outcome = eval(parse(text = paste0("test",k, "$sentiment")))
    CM<-as.matrix(table(prediction, outcome))  # chack with lower values
    print(paste0("Accuracy of cluster ",k))
    
    # count corrects
    for (i in intersect(rownames(CM),colnames(CM))) {
            num_correct = num_correct + CM[i,i]  
        }
    
    # get accuracy
    total_correct = total_correct + num_correct
    Accuracy = num_correct/sum(CM)*100
    print(Accuracy)
    total_predicted = total_predicted + sum(CM)
    
    print(CM)
}
print(sprintf('Total Accuracy: %s',total_correct/total_predicted*100))
